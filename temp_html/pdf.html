import { useEffect } from "react"; function ExamProtection({ isExamActive }) { // 1. Detect tab switch / minimize / another app useEffect(() => { if (!isExamActive) return; const handleVisibility = () => { if (document.hidden) { console.log("User left tab/window"); } }; document.addEventListener( "visibilitychange", handleVisibility ); return () => { document.removeEventListener( "visibilitychange", handleVisibility ); }; }, [isExamActive]); // 2. Prevent refresh / close / typing another URL useEffect(() => { if (!isExamActive) return; const handleBeforeUnload = (e) => { e.preventDefault(); e.returnValue = ""; }; window.addEventListener( "beforeunload", handleBeforeUnload ); return () => { window.removeEventListener( "beforeunload", handleBeforeUnload ); }; }, [isExamActive]); // 3. Detect browser losing focus useEffect(() => { if (!isExamActive) return; const handleBlur = () => { console.log("Browser lost focus"); }; window.addEventListener("blur", handleBlur); return () => { window.removeEventListener( "blur", handleBlur ); }; }, [isExamActive]); // 4. Intercept internal navigation links useEffect(() => { if (!isExamActive) return; const handleClick = (e) => { const anchor = e.target.closest("a"); if (!anchor) return; const confirmed = window.confirm( "Leave exam?" ); if (!confirmed) { e.preventDefault(); } }; document.addEventListener( "click", handleClick ); return () => { document.removeEventListener( "click", handleClick ); }; }, [isExamActive]); // 5. Prevent browser back button useEffect(() => { if (!isExamActive) return; window.history.pushState( null, "", window.location.href ); const handlePopState = () => { const confirmed = window.confirm( "Leave exam?" ); if (!confirmed) { window.history.pushState( null, "", window.location.href ); } }; window.addEventListener( "popstate", handlePopState ); return () => { window.removeEventListener( "popstate", handlePopState ); }; }, [isExamActive]); return null; }
